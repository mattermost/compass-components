Radio buttons are comprised of:

-   Selector
-   Label
    Clicking either the Selector or the Label should toggle the state of the radio button.

export const defaultArgs = {
    label: 'Label',
    size: DEFAULT_RADIO_SIZE,
    checked: false,
    disabled: false,
    hasError: false,
};

export const ControlledRadio = (args) => {
    const [checked, setChecked] = useState(false);
    return (
        <Radio
            {...args}
            onChange={(event) => {
                console.log('#### EVENT:', event);
                setChecked(!checked);
            }}
            checked={checked}
        />
    );
};

export const withChildren = (args) => {
    const newArgs = { ...defaultArgs };
    newArgs.label = (
        <span>
            main label <span style={{ color: 'gray' }}> - details</span>
        </span>
    );
    return newArgs;
};
