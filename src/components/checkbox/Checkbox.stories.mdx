export const defaultArgs = {
    label: 'Default Label',
    size: DEFAULT_CHECKBOX_SIZE,
    checked: false,
    disabled: false,
    hasError: false,
};

export const ControlledCheckbox = (args) => {
    const [checked, setChecked] = useState(false);
    return (
        <Checkbox
            {...args}
            onChange={(event) => {
                console.log('#### EVENT:', event);
                setChecked(!checked);
            }}
            checked={checked}
        />
    );
};
